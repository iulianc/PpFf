package Operators {

  enum OperatorType {
    SOURCE_OP 
    INTERMEDIATE_OP 
    COLLECTOR_OP
  }

  class Operator<T> {
   isSource(): bool
   isCollector(): bool
  }

  class CountOperator<T> {
    svc(): void*
    value(): T::Value
  }

  class MapOperator<T> {
    svc(): void*
  }

  OperatorType .. Operator

  class FooOp

  hide FooOp
  note right of FooOp :  Etc.
  
  Node --> Operator
}

CountOperator --|> Operator
MapOperator --|> Operator
Operator --|> ff_node
