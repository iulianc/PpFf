package Operators {

  enum OperatorType {
    SOURCE_OP 
    INTERMEDIATE_OP 
    COLLECTOR_OP
  }

  class Operator<T> {
   isSource(): bool
   isCollector(): bool
  }

  class MapOperator<T> {
    svc(): void*
  }

  class CountOperator<T> {
    svc(): void*
    value(): T::Value
  }

  OperatorType . Operator


  note right of CountOperator :  Etc.
  
  note right of MapOperator :  Etc.

  Node --> Operator
}

MapOperator --|> Operator
CountOperator --|> Operator
Operator --|> ff_node
