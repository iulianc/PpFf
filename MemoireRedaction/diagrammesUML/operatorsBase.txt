package Operators {

  abstract class BaseOperator {
    {abstract} <<virtual>> svc(): void*
    +=(): BaseOperator
  }

  class CountOperator<T> {
    svc(): void*
    value(): T::Value
  }

  class MapOperator<T> {
    svc(): void*
  }

  note right of CountOperator :  Etc.
  note right of MapOperator :  Etc.

  CountOperator --|> BaseOperator
  MapOperator --|> BaseOperator
}
